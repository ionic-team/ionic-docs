## Usage

### React

### Angular

```tsx
import { DocumentPicker } from '@awesome-cordova-plugins/document-picker/ngx';

constructor(private docPicker: DocumentPicker) { }

...

this.docPicker.getFile('all')
  .then(uri => console.log(uri))
  .catch(e => console.log(e));

```
